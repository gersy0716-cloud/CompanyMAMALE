---

## ⚠️ BaseMulti 注意事项

1. **系统字段不传递**：id, created_at 等由系统自动管理
2. **响应格式**：查询返回 `{data: [...]}` 而非 `{records: [...]}`
3. **字段访问**：直接用 `record.url`，不用 `record.fields.url`
4. **日期格式**：使用 ISO 8601（`YYYY-MM-DDTHH:mm:ssZ`）
5. **长文本存JSON**：BaseMulti 不支持 JSON 类型，用 Long Text + `JSON.stringify()`

## 🔍 错误检验与调试

### 常见错误类型

#### 1. HTTP状态码错误

```javascript
// ❌ 错误示例：未检查HTTP状态
const response = await fetch(url);
const data = await response.json();

// ✅ 正确示例：完整的错误检查
const response = await fetch(url, {
    method: 'POST',
    headers: {
        'x-bm-token': API_TOKEN,
        'Content-Type': 'application/json'
    },
    body: JSON.stringify(data)
});

// 检查HTTP状态
if (!response.ok) {
    // 尝试解析错误响应
    let errorData;
    const contentType = response.headers.get('content-type');

    if (contentType && contentType.includes('application/json')) {
        try {
            errorData = await response.json();
            throw new Error(`API错误 (${response.status}): ${errorData.error || errorData.message || '未知错误'}`);
        } catch (e) {
            throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
    } else {
        const text = await response.text();
        throw new Error(`HTTP ${response.status}: ${text.substring(0, 200)}`);
    }
}
```

#### 2. JSON解析错误

```javascript
// ✅ 正确示例：安全的JSON解析
async function apiRequest(url, method, body = null) {
    const headers = {
        'x-bm-token': API_TOKEN
    };
    if (body) {
        headers['Content-Type'] = 'application/json';
    }

    const response = await fetch(url, {
        method,
        headers,
        body: body ? JSON.stringify(body) : null
    });

    if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }

    // 检查响应类型
    const contentType = response.headers.get('content-type');
    if (contentType && contentType.includes('application/json')) {
        try {
            return await response.json();
        } catch (e) {
            throw new Error(`JSON解析失败: ${e.message}`);
        }
    } else {
        // 如果不是JSON，尝试解析文本
        const text = await response.text();
        try {
            return JSON.parse(text);
        } catch (e) {
            throw new Error(`服务器返回了非JSON响应: ${text.substring(0, 200)}`);
        }
    }
}
```

#### 3. 系统字段过滤错误

```javascript
// ❌ 错误示例：传递了系统字段
const data = {
    id: 123,  // ❌ 系统字段，不应传递
    name: '测试',
    created_at: new Date().toISOString()  // ❌ 系统字段
};
await apiRequest('/records', 'POST', data);

// ✅ 正确示例：过滤系统字段
function filterSystemFields(data) {
    const systemFields = ['id', 'created_at', 'updated_at', 'created_by', 'updated_by'];
    const filtered = { ...data };

    systemFields.forEach(field => {
        if (field in filtered) {
            console.log(`⚠️ 过滤系统字段: ${field}`);
            delete filtered[field];
        }
    });

    return filtered;
}

// 使用
const data = filterSystemFields({
    id: 123,
    name: '测试',
    created_at: new Date().toISOString()
});
await apiRequest('/records', 'POST', data);
```

#### 4. 分页参数错误

```javascript
// ✅ 正确示例：构建分页URL
async function fetchRecords(tableName, page = 1, pageLimit = 10, sortField = null) {
    const url = new URL(`${BASE_URL}/api/bases/${BASE_ID}/tables/${tableName}/records`);

    // 添加分页参数
    url.searchParams.append('page', page);
    url.searchParams.append('pageLimit', pageLimit);

    // 添加排序参数
    if (sortField) {
        const filters = [["orderBy", [sortField, "desc"]]];
        url.searchParams.append('filters', btoa(JSON.stringify(filters)));
    }

    const response = await apiRequest(url.toString(), 'GET');

    // 验证响应格式
    if (!response.data) {
        throw new Error('响应格式错误：缺少data字段');
    }

    return {
        data: response.data,
        current_page: response.current_page || 1,
        last_page: response.last_page || 1,
        total: response.total || 0,
        per_page: response.per_page || pageLimit
    };
}
```

### 错误处理最佳实践

```javascript
// ✅ 完整的错误处理示例
async function createRecord(tableId, data) {
    try {
        console.log('💾 [Database] 创建记录 - 表ID:', tableId);

        // 1. 参数验证
        if (!tableId) {
            throw new Error('表ID不能为空');
        }
        if (!data || typeof data !== 'object') {
            throw new Error('数据格式错误');
        }

        // 2. 过滤系统字段
        const filteredData = filterSystemFields(data);

        // 3. 发送请求
        const response = await fetch(`${BASE_URL}/api/bases/${BASE_ID}/tables/${tableId}/records`, {
            method: 'POST',
            headers: {
                'x-bm-token': API_TOKEN,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(filteredData)
        });

        // 4. 检查HTTP状态
        if (!response.ok) {
            const errorText = await response.text();
            throw new Error(`HTTP ${response.status}: ${errorText}`);
        }

        // 5. 解析响应
        const record = await response.json();

        // 6. 验证响应格式
        if (!record.id) {
            throw new Error('响应格式错误：缺少id字段');
        }

        console.log('✅ [Database] 创建成功 - 记录ID:', record.id);
        return record;

    } catch (error) {
        // 7. 错误分类处理
        console.error('❌ [Database] 创建失败:', error);

        // 网络错误
        if (error.name === 'TypeError' && error.message.includes('fetch')) {
            throw new Error('网络连接失败，请检查网络后重试');
        }

        // HTTP错误
        if (error.message.includes('HTTP 401')) {
            throw new Error('认证失败，请检查API Key');
        }
        if (error.message.includes('HTTP 404')) {
            throw new Error('表不存在，请检查表ID');
        }
        if (error.message.includes('HTTP 500')) {
            throw new Error('服务器错误，请稍后重试');
        }

        // 其他错误
        throw error;
    }
}
```

function validateResponse(response, requiredFields = []) {
    if (!response) {
        throw new Error('响应为空');
    }

    for (const field of requiredFields) {
        if (!(field in response)) {
            throw new Error(`响应缺少必需字段: ${field}`);
        }
    }

    return true;
}
